package comp3111.covid;

import javafx.beans.property.SimpleStringProperty;
/**
 * A Class to contain key information in one object
 * Class includes functions to set the variable and retrieve the information
 */
public class Row {
	private final SimpleStringProperty country;
    private final SimpleStringProperty date;
    private Long totalCases = (long) -1;
    private Double totalCasesPerM = -1.0;
    private Long totalDeaths = (long) 0;
    private Double totalDeathsPerM = -1.0;
    private Long totalVaccinated = (long) 0;
    private Double vaccinationRate = -1.0;
    
    /**
     * Default Constructor
     * This constructor creates Row instance with country and date attributes.
     * @param _country String type country name.
     * @param _date String type date in "M/d/yyyy" format.
     */
    public Row(String _country, String _date) {
    	this.country = new SimpleStringProperty(_country);
        this.date = new SimpleStringProperty(_date);
        
	}
    
    /**
     * This function returns instances' country attribute
     * @return Return a String type country attribute.
     */
    public String getCountry() {
        return country.get();
    }
    
    /**
     * This function set instances' country attribute.
     * @param _country String type country name.
     */
    public void setCountry(String _country) {
        country.set(_country);
    }
    
    /**
     * This function returns instances' date attribute
     * @return Return a String type date attribute in "M/d/yyyy" format.
     */
    public String getDate() {
        return date.get();
    }
    
    /**
     * This function set instances' date attribute.
     * @param _date String type date in "M/d/yyyy" format.
     */
    public void setDate(String _date) {
        country.set(_date);
    }
    
    /**
     * This function returns total cases.
     * @return Return a Long type of total cases.
     */
    public Long getTotalCases() {
        return this.totalCases;
    }
    
    /**
     * This function set instances' _totalCases attribute.
     * @param _totalCases Long type.
     */
    public void setTotalCases(Long _totalCases) {
        totalCases = _totalCases;
    }
    
    /**
     * This function returns total cases per million.
     * @return Return a Double type of total cases per million.
     */
    public Double getTotalCasesPerM() {
    	return this.totalCasesPerM;
    }
    
    /**
     * This function set instances' _totalCasesPerM attribute.
     * @param _totalCasesPerM Double type.
     */
    public void setTotalCasesPerM(Double _totalCasesPerM) {
    	totalCasesPerM = _totalCasesPerM;
    }
    
    /**
     * This function returns total deaths.
     * @return Return a Long type of total deaths.
     */
    public Long getTotalDeaths() {
        return this.totalDeaths;
    }

    /**
     * This function set instances' _totalDeaths attribute.
     * @param _totalDeaths Long type.
     */
    public void setTotalDeaths(Long _totalDeaths) {
        totalDeaths = _totalDeaths;
    }
    
    /**
     * This function returns total deaths per million.
     * @return Return a Double type of total deaths per million.
     */
    public Double getTotalDeathsPerM() {
    	return this.totalDeathsPerM;
    }
    
    /**
     * This function set instances' _totalDeathsPerM attribute.
     * @param _totalDeathsPerM Double type.
     */
    public void setTotalDeathsPerM(Double _totalDeathsPerM) {
    	totalDeathsPerM = _totalDeathsPerM;
    }
    
    /**
     * This function returns total fully vaccinated.
     * @return Return a Long type of total fully vaccinated.
     */
    public Long getTotalVaccinated() {
        return this.totalVaccinated;
    }
    
    /**
     * This function set instances' _totalVaccinated attribute.
     * @param _totalVaccinated Long type.
     */
    public void setTotalVaccinated(Long _totalVaccinated) {
    	totalVaccinated = _totalVaccinated;
    }
    
    /**
     * This function returns rate of fully vaccinated.
     * @return Return a Double type rate of fully vaccinated.
     */
    public Double getVaccinationRate() {
    	return this.vaccinationRate;
    }
    
    /**
     * This function set instances' vaccinationRate attribute.
     * @param fullyVaccinated Long type of total fully vaccinated count.
     * @param population Double type of population count.
     */
    public void setVaccinationRate(Long fullyVaccinated, Double population) {
    	if(fullyVaccinated == -1) {
    		vaccinationRate  = (double) -1;
    	}
    	else {
        	vaccinationRate = 100*fullyVaccinated/population;
    	}
    }
    /**
     * This functions is for generating Tables only
     * This function converts a null output from total cases into a string saying "No record on this date" or the total cases values in String type otherwise.
     * @return Return a string to be output on the table 
     */
    public String getTotalCasesTable() {
        if(this.totalCases == -1) {
        	return "No record on this date";
        }
        else {
        	return this.totalCases.toString();
        }
    }
    /**
     * This functions is for generating Tables only
     * This function converts a null output from totalCasesPerM into a string saying "No record on this date" or the total cases values in String type otherwise.
     * @return Return a string to be output on the table 
     */
    public String getTotalCasesPerMTable() {
        if(this.totalCasesPerM == -1) {
        	return "No record on this date";
        }
        else {
        	return this.totalCasesPerM.toString();
        }
    }
    /**
     * This functions is for generating Tables only
     * This function converts a null output from total deaths into a string saying "No record on this date" or the total cases values in String type otherwise.
     * @return Return a string to be output on the table 
     */
    public String getTotalDeathsTable() {
        if(this.totalDeaths == -1) {
        	return "No record on this date";
        }
        else {
        	return this.totalDeaths.toString();
        }
    }
    /**
     * This functions is for generating Tables only
     * This function converts a null output from totalDeathsPerM into a string saying "No record on this date" or the total cases values in String type otherwise.
     * @return Return a string to be output on the table 
     */
    public String getTotalDeathsPerMTable() {
        if(this.totalDeathsPerM == -1) {
        	return "No record on this date";
        }
        else {
        	return this.totalDeathsPerM.toString();
        }
    }
    /**
     * This functions is for generating Tables only
     * This function converts a null output from total vaccinated into a string saying "No record on this date" or the total cases values in String type otherwise.
     * @return Return a string to be output on the table 
     */
    public String getTotalVaccinatedTable() {
        if(this.totalVaccinated == -1) {
        	return "No record on this date";
        }
        else {
        	return this.totalVaccinated.toString();
        }
    }
    /**
     * This functions is for generating Tables only
     * This function converts a null output from vaccination rate into a string saying "No record on this date" or the total cases values in String type otherwise.
     * @return Return a string to be output on the table 
     */
    public String getVaccinationRateTable() {
        if(this.vaccinationRate == -1) {
        	return "No record on this date";
        }
        else {
        	return this.vaccinationRate.toString();
        }
    }
    
}
